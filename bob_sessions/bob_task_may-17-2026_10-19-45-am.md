259 |   key: string,
260 |   ttlMs: number,
261 |   force: boolean,
262 |   fetchFresh: () => Promise<T>,
263 | ): Promise<T> {
264 |   const cache = remoteCache();

267 |     !force &&

289 | function revalidateRemoteCache<T>(
290 |   key: string,
291 |   fetchFresh: () => Promise<T>,
292 | ): void {
293 |   if (revalidatingRemoteKeys.has(key)) return;
294 |   revalidatingRemoteKeys.add(key);
295 |   void fetchFresh()
296 |     .then((value) => {
297 |       const cache = remoteCache();
298 |       storeLib.setStoreValue(
299 |         REMOTE_DATA_CACHE_KEY,
300 |         pruneRemoteCache({
301 |           ...cache,
302 |           [key]: { value, timestamp: Date.now() },
303 |         }),
304 |       );
305 |     })
306 |     .catch(() => {
307 |       // Stale cache remains usable; transient provider failures are surfaced on forced refresh.
308 |     })
309 |     .finally(() => {
310 |       revalidatingRemoteKeys.delete(key);
311 |     });

314 | function pruneRemoteCache(
315 |   cache: Record<string, unknown>,
316 | ): Record<string, unknown> {
317 |   const now = Date.now();
318 |   const entries = Object.entries(cache)
319 |     .filter(([, entry]) => {
320 |       if (!entry || typeof entry !== "object" || Array.isArray(entry))
321 |         return false;
322 |       const timestamp = (entry as Partial<RemoteCacheEntry<unknown>>).timestamp;
323 |       return (
324 |         typeof timestamp === "number" &&
325 |         now - timestamp <= REMOTE_DETAIL_TTL_MS * 4
326 |       );
327 |     })
328 |     .sort((a, b) => {
329 |       const aTimestamp =
330 |         (a[1] as Partial<RemoteCacheEntry<unknown>>).timestamp ?? 0;
331 |       const bTimestamp =
332 |         (b[1] as Partial<RemoteCacheEntry<unknown>>).timestamp ?? 0;
333 |       return bTimestamp - aTimestamp;
334 |     })
335 |     .slice(0, MAX_REMOTE_CACHE_ENTRIES);
336 |   return Object.fromEntries(entries);
337 | }

383 |   let nextGitMessageId = 0;

## 2. IPC Communication Flow

**File:** `overcode-ipc-flow.excalidraw`

### Overview
Detailed view of how Inter-Process Communication (IPC) works in Overcode.

### The IPC Pattern

```
Renderer → Preload → Main → Preload → Renderer
```

#### Step-by-Step Flow

1. **Renderer Calls API**
   ```typescript
   const repos = await window.api.github.repos();
   ```

2. **Preload Forwards Request**
   ```typescript
   github: {
     repos: () => ipcRenderer.invoke('github:repos')
   }
   ```

3. **Main Handles Request**
   ```typescript
   ipcMain.handle('github:repos', async () => {
     return github.listRepos(githubToken());
   });
   ```

4. **Response Returns**
   - Main resolves promise
   - Preload forwards result
   - Renderer receives data

### IPC Channel Groups

#### auth (3 channels)
- `auth:connect` - OAuth flow
- `auth:disconnect` - Remove credentials
- `auth:status` - Check connection state

#### git (12 channels)
- `git:scan` - Scan directories for repos
- `git:status` - Get repo status
- `git:log` - Get commit history
- `git:stashes` - List stashes
- `git:stash-show` - Show stash diff
- `git:worktrees` - List worktrees
- `git:file` - Read file content
- `git:divergence` - Check branch divergence
- `git:push` - Push to remote
- `git:pull` - Pull from remote
- `git:commit` - Create commit
- `git:stash-pop` - Apply and remove stash
- `git:stash-drop` - Remove stash

#### github (4 channels)
- `github:repos` - List repositories
- `github:issues` - List/get issues
- `github:prs` - List/get pull requests
- `github:pipelines` - List workflow runs
- `github:comment` - Add PR comment

#### gitlab (4 channels)
- `gitlab:projects` - List projects
- `gitlab:mrs` - List/get merge requests
- `gitlab:issues` - List/get issues
- `gitlab:pipelines` - List pipelines
- `gitlab:comment` - Add MR comment

#### ai (2 channels)
- `ai:complete` - Call watsonx.ai Granite
- `ai:status` - Check AI configuration

#### store (3 channels)
- `store:get` - Read from electron-store
- `store:set` - Write to electron-store
- `store:list` - List all keys

- Renderer never accesses Node.js APIs directly

## 4. Data Flow

**File:** `overcode-data-flow.excalidraw`

### Overview
Maps how data moves through the Overcode system from filesystem to UI.

### Data Sources

#### 1. Local Filesystem
- Git repositories
- Working directories
- `.git` folders
- Configuration files

#### 2. Git Worker (Utility Process)
- Reads git data via simple-git
- Watches filesystem via chokidar
- Detects changes with 500ms debounce
- Forwards to main process

#### 3. electron-store (Persistence)
Stores 5 key data types:
- **repositories**: List of scanned repos with metadata
- **activity**: Recent activity feed (500 items max)
- **settings**: User preferences + AI config
- **remote_data_cache**: Cached API responses (200 entries max)
- **ai_audit_log**: AI feature usage logs (120 entries max)

#### 4. External APIs
- **GitHub API**: Via Octokit client
- **GitLab API**: Via fetch
- **watsonx.ai Granite**: Via REST API

### Data Flow Patterns

#### Pattern 1: Local Git Data
```
Filesystem → Git Worker → Main Process → electron-store → Renderer
```

**Example**: Scanning repositories
1. User triggers scan
2. Git Worker reads directories
3. Finds git repos, reads status
4. Sends to main process
5. Main stores in electron-store
6. Renderer displays list

#### Pattern 2: Remote API Data (Cache-First)
```
Renderer → Main → electron-store (cache) → Renderer
                ↓ (background)
           External API → electron-store (update)
```

**Example**: Loading GitHub PRs
1. Renderer requests PRs
2. Main checks cache (2min TTL for lists)
3. Returns cached data immediately
4. Fetches fresh data in background
5. Updates cache
6. Renderer auto-refreshes on next request

#### Pattern 3: AI Feature
```
Renderer → Main → watsonx.ai Granite → Main → Renderer
                                      ↓
                                electron-store (audit log)
```

**Example**: Generating commit message
1. Renderer sends staged diff
2. Main builds prompts
3. Calls watsonx.ai Granite
4. Logs to audit trail
5. Returns generated message
6. Renderer displays in UI

### Caching Strategy

#### Remote Data Cache
- **List endpoints**: 2min TTL (repos, PRs, issues)
- **Detail endpoints**: 5min TTL (PR detail, issue detail)
- **Max entries**: 200 (LRU eviction)
- **Stale-while-revalidate**: Serve cache, fetch fresh in background

#### Cache Keys
```
github:repos
github:prs:{owner}/{repo}
github:pr-detail:{owner}/{repo}:{number}
github:issues:all
github:issues:{owner}/{repo}
gitlab:projects
gitlab:mrs:{projectId}
gitlab:mr-detail:{projectId}:{iid}
```

### Performance Optimizations

1. **Virtual Lists**: react-window with 64px item height
2. **Pagination**: Load 100 commits at a time
3. **Debouncing**: 500ms for filesystem changes
4. **Cache-First**: Instant UI, background refresh
5. **Lazy Loading**: Load data only when screen is active
